            # 快速使用

            Request 库实例化了一个简单的请求函数,方便快捷的使用.不过,建议大家使用自定义使用.

            import { zRequest } from "@zoouse/request";
zRequest({
  baseURL: "http://127.0.0.1:4523/mock/1391418",
  url: "/all",
  interceptors: {
    requestInterceptor(config) {
      console.log("请求拦截器");
      return config;
    },
    responseInterceptor(res) {
      console.log("响应拦截器");
      return res;
    },
  },
})
  .then((res) => {
    console.log("res", res);
  })
  .catch((err) => {
    console.log("err", err);
  });
