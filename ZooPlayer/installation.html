              # 原生JS使用

              # Packges

              使用npm或yarn安装

              然后引入到项目中即可

              import ZPlayer from 'zooplayer'

const player = new ZPlayer({
    el: "", // 指定挂载的选择器
    video: {
        src: "" // 视频播放地址
    }
})

player.mount()

              # CDN

              引入

              <script src="https://unpkg.com/zooplayer@lastest/dist/zooplayer.js"></script>

              使用

              <script>
const player = new ZPlayer({
    el: "#app",
    video: {
        src: 'http://clips.vorwaerts-gmbh.de/big_buck_bunny.mp4'
    }
})

player.mount()
</script>

              import ZPlayer from "./player"

Vue.use(ZPlayer);
